```php

use Polar;
use Polar\Models\Components;

    ->setSecurity(
        '<YOUR_BEARER_TOKEN_HERE>'
    )
    ->build();

    url: 'https://webhook.site/cb791d80-f26e-4f8c-be88-6e56054192b0',

    request: $request
